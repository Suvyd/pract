#include <stdio.h>

int main() {
    printf("Bohdan Suvyd\n");
    int t1, t2, t3;
    float v1, v2, v3, v_total, t_total;

    // Зчитуємо значення t1, t2 і t3
    scanf("%d %d %d", &t1, &t2, &t3);

    // Обчислюємо швидкість кожного гостя
    v1 = 1.0 / t1;
    v2 = 1.0 / t2;
    v3 = 1.0 / t3;

    // Обчислюємо загальну швидкість всіх гостей разом
    v_total = v1 + v2 + v3;

    // Обчислюємо час, необхідний гостям для з'їдання одного пирога
    t_total = 1.0 / v_total;

    // Виводимо результат
    printf("%.2f\n", t_total);

    return 0;
}
